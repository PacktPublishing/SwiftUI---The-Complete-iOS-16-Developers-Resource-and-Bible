import SwiftUI
import CloudKit

struct ShowPictureView: View {
   @EnvironmentObject var appData: ApplicationData
   let selectedCity: CityViewModel

   var body: some View {
      VStack {
         Image(uiImage: selectedCity.cityPicture)
            .resizable()
            .scaledToFit()
         Spacer()
      }.navigationBarTitle(selectedCity.cityName)
   }
}
