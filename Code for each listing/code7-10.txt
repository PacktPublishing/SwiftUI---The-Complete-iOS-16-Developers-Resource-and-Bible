struct ContentView: View {
   @EnvironmentObject var appData: ApplicationData

   let guides = [
      GridItem(.fixed(75)),
      GridItem(.fixed(75)),
      GridItem(.fixed(75))
   ]
   var body: some View {
      ScrollView {
         LazyVGrid(columns: guides) {
            ForEach(appData.userData) { book in
               Image(book.cover)
                  .resizable()
                  .scaledToFit()
            }
         }
      }.padding()
   }
}
